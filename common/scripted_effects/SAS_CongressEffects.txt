# Generates the GUI for the House of Representatives
SAS_RepresentativesReload_effect = {
	clear_array = representatives_main_array

	for_each_loop = {
		array = representatives_seat_array

		add_to_array = { temp_seat = representatives_seat_array^i }
	}

	set_temp_variable = { temp_index_1 = 0 }
	set_temp_variable = { temp_index_2 = 1 }

	set_temp_variable = { temp_size = temp_seat^num }
	subtract_from_temp_variable = { temp_size = 1 }

	while_loop_effect = {
		limit = {
			check_variable = { temp_index_1 < temp_size }
		}
		add_to_variable = { temp_seat^temp_index_2 = temp_seat^temp_index_1 }
		add_to_temp_variable = { temp_index_1 = 1 }
		add_to_temp_variable = { temp_index_2 = 1 }
	}

	for_loop_effect = {
		start = 0
		end = temp_seat^num

		resize_array = { array = representatives_main_array value = representatives_member_array^v size = temp_seat^v }
	}

	clear_array = temp_seat
}

# Effect to udpate the representative seats - optional
SAS_ModifyRepresentativeSeat_effect = {
	subtract_from_temp_variable = { temp_member = 1 }

	set_temp_variable = { temp = representatives_seat_array^temp_member }
	add_to_temp_variable = { temp = temp_seats }

	if = {
		limit = {
			check_variable = { temp > representatives_main_array^num }
		}
		subtract_from_temp_variable = { temp = representatives_main_array^num }
		subtract_from_temp_variable = { temp_seats = temp }
	}
	else_if = {
		limit = {
			check_variable = { temp < 0 }
		}
		multiply_temp_variable = { temp = -1 }
		subtract_from_temp_variable = { temp_seats = temp }
	}

	for_loop_effect = {
		start = 0
		end = representatives_seat_array^num

		add_to_temp_array = { temp_array = v }
	}

	remove_from_temp_array = { array = temp_array value = temp_member }

	add_to_variable = { representatives_seat_array^temp_member = temp_seats }

	if = {
		limit = {
			check_variable = { temp_seats > 0 }
		}
		while_loop_effect = {
			limit = {
				check_variable = { temp_seats > 0 }
			}
			for_loop_effect = {
				start = 0
				end = temp_array^num

				if = {
					limit = {
						check_variable = { representatives_seat_array^temp_array^v > 0 }
						check_variable = { temp_seats > 0 } 
					}
					subtract_from_variable = { representatives_seat_array^temp_array^v = 1 } 
					subtract_from_temp_variable = { temp_seats = 1 } 
				}
			}
		}
	}
	else_if = {
		limit = {
			check_variable = { temp_seats < 0 }
		}
		while_loop_effect = {
			limit = {
				check_variable = { temp_seats < 0 }
			}
			for_loop_effect = {
				start = 0
				end = temp_array^num

				if = {
					limit = {
						check_variable = { representatives_seat_array^temp_array^v < representatives_main_array^num } 
						check_variable = { temp_seats < 0 } 
					}
					add_to_variable = { representatives_seat_array^temp_array^v = 1 } 
					add_to_temp_variable = { temp_seats = 1 } 
				}
			}
		}
	}
	SAS_RepresentativesReload_effect = yes
}

# Effect to add DEMOCRATIC support in the house of representatives. Up to 5 democrats move from the opposition to supporting the government.
SAS_House_IncreaseDemSupport_small_effect = {
	custom_effect_tooltip = SAS_House_IncreaseDemSupport_small_tt
	# First grab the current values for the seats
	set_temp_variable = {
		democratic_opposition_tempvar = representatives_seat_array^0
	}
	set_temp_variable = {
		democratic_support_tempvar = representatives_seat_array^1
	}
	set_temp_variable = {
		republican_support_tempvar = representatives_seat_array^2
	}
	set_temp_variable = {
		republican_opposition_tempvar = representatives_seat_array^3
	}
	# If there's AT LEAST five un-supportive democrats
	if = {
		limit = {
			check_variable = {
				democratic_opposition_tempvar > 4
			}
		}
		add_to_temp_variable = {
			democratic_support_tempvar = 5
		}
		subtract_from_temp_variable = {
			democratic_opposition_tempvar = 5
		}
	}
	# Otherwise
	else = {
		add_to_temp_variable = {
			democratic_support_tempvar = democratic_opposition_tempvar
		}
		set_temp_variable = {
			democratic_opposition_tempvar = 0
		}
	}
	# Clear the arrays, and rebuild them with the new values
	SAS_House_CongressClearArrays_effect = yes

	add_to_array = { representatives_seat_array = democratic_opposition_tempvar } # Democrats (opposition)
	add_to_array = { representatives_seat_array = democratic_support_tempvar } # Democrats (support)
	add_to_array = { representatives_seat_array = republican_support_tempvar }  # Republicans (support)
	add_to_array = { representatives_seat_array = republican_opposition_tempvar } # Republicans (opposition)

	if = {
		limit = {
			OR = {
				has_country_leader_ideology = liberalism
				has_country_leader_ideology = socialism
			}
		}
		custom_effect_tooltip = SAS_House_IncreaseDemocraticSeats_loyal_tt
	}
	else = {
		custom_effect_tooltip = SAS_House_IncreaseDemocraticSeats_notloyal_tt
	}
	# First grab the current values for the seats
	set_temp_variable = {
		democratic_opposition_tempvar = representatives_seat_array^0
	}
	set_temp_variable = {
		democratic_support_tempvar = representatives_seat_array^1
	}
	set_temp_variable = {
		republican_support_tempvar = representatives_seat_array^2
	}
	set_temp_variable = {
		republican_opposition_tempvar = representatives_seat_array^3
	}
	# The new seats will ALWAYS support the government IF the president is democratic. Otherwise, they won't.
	if = {
		limit = {
			OR = {
				has_country_leader_ideology = liberalism
				has_country_leader_ideology = socialism
			}
		}
		add_to_temp_variable = {
			democratic_support_tempvar = seat_change_tempvar
		}
	}
	else = {
		add_to_temp_variable = {
			democratic_opposition_tempvar = seat_change_tempvar
		}
	}
	# Hand the change in republicans seats - remove FIRST from opposition then into support
	if = {
		limit = {
			OR = {
				check_variable = {
					republican_opposition_tempvar > seat_change_tempvar
				}
				check_variable = {
					republican_opposition_tempvar = seat_change_tempvar
				}
			}
		}
		subtract_from_temp_variable = {
			republican_opposition_tempvar = seat_change_tempvar
		}
	}
	else = {
		subtract_from_temp_variable = {
			seat_change_tempvar = republican_opposition_tempvar
		}
		set_temp_variable = {
			republican_opposition_tempvar = 0
		}
		subtract_from_temp_variable = {
			republican_support_tempvar = seat_change_tempvar
		}
	}

	# Clear the arrays, and rebuild them with the new values
	clear_array = representatives_seat_array

	add_to_array = { representatives_seat_array = democratic_opposition_tempvar } # Democrats (opposition)
	add_to_array = { representatives_seat_array = democratic_support_tempvar } # Democrats (support)
	add_to_array = { representatives_seat_array = republican_support_tempvar }  # Republicans (support)
	add_to_array = { representatives_seat_array = republican_opposition_tempvar } # Republicans (opposition)

# Effect to increase democratic seats in the senate
SAS_Senate_IncreaseDemocraticSeats_effect = {
	if = {
		limit = {
			OR = {
				has_country_leader_ideology = liberalism
				has_country_leader_ideology = socialism
			}
		}
		custom_effect_tooltip = SAS_Senate_IncreaseDemocraticSeats_loyal_tt
	}
	else = {
		custom_effect_tooltip = SAS_Senate_IncreaseDemocraticSeats_notloyal_tt
	}
	# First grab the current values for the seats
	set_temp_variable = {
		democratic_opposition_tempvar = senate_seat_array^0
	}
	set_temp_variable = {
		democratic_support_tempvar = senate_seat_array^1
	}
	set_temp_variable = {
		republican_support_tempvar = senate_seat_array^2
	}
	set_temp_variable = {
		republican_opposition_tempvar = senate_seat_array^3
	}
	# The new seats will ALWAYS support the government IF the president is democratic. Otherwise, they won't.
	if = {
		limit = {
			OR = {
				has_country_leader_ideology = liberalism
				has_country_leader_ideology = socialism
			}
		}
		add_to_temp_variable = {
			democratic_support_tempvar = seat_change_tempvar
		}
	}
	else = {
		add_to_temp_variable = {
			democratic_opposition_tempvar = seat_change_tempvar
		}
	}
	# Hand the change in republicans seats - remove FIRST from opposition then into support
	if = {
		limit = {
			OR = {
				check_variable = {
					republican_opposition_tempvar > seat_change_tempvar
				}
				check_variable = {
					republican_opposition_tempvar = seat_change_tempvar
				}
			}
		}
		subtract_from_temp_variable = {
			republican_opposition_tempvar = seat_change_tempvar
		}
	}
	else = {
		subtract_from_temp_variable = {
			seat_change_tempvar = republican_opposition_tempvar
		}
		set_temp_variable = {
			republican_opposition_tempvar = 0
		}
		subtract_from_temp_variable = {
			republican_support_tempvar = seat_change_tempvar
		}
	}

	# Clear the arrays, and rebuild them with the new values
	clear_array = representatives_seat_array

	add_to_array = { senate_seat_array = democratic_opposition_tempvar } # Democrats (opposition)
	add_to_array = { senate_seat_array = democratic_support_tempvar } # Democrats (support)
	add_to_array = { senate_seat_array = republican_support_tempvar }  # Republicans (support)
	add_to_array = { senate_seat_array = republican_opposition_tempvar } # Republicans (opposition)

	# Re-Load the UI
	SAS_reloadCongress_effect = yes
}
